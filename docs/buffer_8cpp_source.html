25BufMgr::BufMgr(std::uint32_t bufs)

56void BufMgr::allocBuf(FrameId& frame) {
57 // perform first part of clock algorithm to search for
58 // open buffer frame
59 // Assumes non-concurrent access to buffer manager
60 std::uint32_t numScanned = 0;
61 bool found = 0;
62
63 while (numScanned < 2 * numBufs) // Need to scn twice
64 {
65 // advance the clock
66 advanceClock();
67 numScanned++;
68
69 // if invalid, use frame
70 if (!bufDescTable[clockHand].valid) {
71 break;
72 }
73
74 // is valid, check referenced bit
75 if (!bufDescTable[clockHand].refbit) {
76 // check to see if someone has it pinned
77 if (bufDescTable[clockHand].pinCnt == 0) {
78 // hasn't been referenced and is not pinned, use it
79 // remove previous entry from hash table
80 hashTable->remove(bufDescTable[clockHand].file, bufDescTable[clockHand].pageNo);
81 found = true;
82 break;
83 }
84 } else {
85 // has been referenced, clear the bit
86 bufStats.accesses++;
87 bufDescTable[clockHand].refbit = false;
88 }
89 }
90
91 // check for full buffer pool
92 if (!found && numScanned >= 2 * numBufs) {
93 throw BufferExceededException();
94 }
95
96 // flush any existing changes to disk if necessary
97 if (bufDescTable[clockHand].dirty) {
98 bufStats.diskwrites++;
99 // status = bufDescTable[clockHand].file->writePage(bufDescTable[clockHand].pageNo,
100 bufDescTable[clockHand].file->writePage(bufDescTable[clockHand].pageNo, bufPool[clockHand]);
101 }
102
103 // Reset all the BufDesc entry for the frame before returning the frame
104 bufDescTable[clockHand].Clear();
105
106 // return new frame number
107 frame = clockHand;
108} // end allocBuf

110void BufMgr::readPage(File* file, const PageId pageNo, Page*& page) {
111 // check to see if it is already in the buffer pool
112 // std::cout << "readPage called on file.page " << file << "." << pageNo << endl;
113 FrameId frameNo = 0;
114 try {
115 hashTable->lookup(file, pageNo, frameNo);
116
117 // set the referenced bit
118 bufDescTable[frameNo].refbit = true;
119 bufDescTable[frameNo].pinCnt++;
120 page = &bufPool[frameNo];
121 } catch (const HashNotFoundException& e) // not in the buffer pool, must allocate a new page
122 {
123 // alloc a new frame
124 allocBuf(frameNo);
125
126 // read the page into the new frame
127 bufStats.diskreads++;
128 // status = file->readPage(pageNo, &bufPool[frameNo]);
129 bufPool[frameNo] = file->readPage(pageNo);
130
131 // set up the entry properly
132 bufDescTable[frameNo].Set(file, pageNo);
133 page = &bufPool[frameNo];
134
135 // insert in the hash table
136 hashTable->insert(file, pageNo, frameNo);
137 }

140void BufMgr::unPinPage(File* file, const PageId pageNo, const bool dirty) {
141 // lookup in hashtable
142 FrameId frameNo = 0;
143 hashTable->lookup(file, pageNo, frameNo);
144
145 if (dirty == true) bufDescTable[frameNo].dirty = dirty;
146
147 // make sure the page is actually pinned
148 if (bufDescTable[frameNo].pinCnt == 0) {
149 throw PageNotPinnedException(file->filename(), pageNo, frameNo);
150 } else
151 bufDescTable[frameNo].pinCnt--;

172void BufMgr::flushFile(const File* file) {
173 for (std::uint32_t i = 0; i < numBufs; i++) {
174 BufDesc* tmpbuf = &(bufDescTable[i]);
175 if (tmpbuf->file && tmpbuf->valid == true && tmpbuf->file == file) {
176 if (tmpbuf->pinCnt > 0)
177 throw PagePinnedException(file->filename(), tmpbuf->pageNo, tmpbuf->frameNo);
178
179 if (tmpbuf->dirty == true) {
180 // if ((status = tmpbuf->file->writePage(tmpbuf->pageNo, &(bufPool[i]))) != OK)
181 tmpbuf->file->writePage(tmpbuf->pageNo, bufPool[i]);
182 tmpbuf->dirty = false;
183 }
184
185 hashTable->remove(file, tmpbuf->pageNo);
186 tmpbuf->Clear();
187 } else if (tmpbuf->valid == false && tmpbuf->file == file)
188 throw BadBufferException(tmpbuf->frameNo, tmpbuf->dirty, tmpbuf->valid, tmpbuf->refbit);
189 }
190}
