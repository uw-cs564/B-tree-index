28BTreeIndex::BTreeIndex(const std::string &relationName,
29 std::string &outIndexName,
30 BufMgr *bufMgrIn,
31 const int attrByteOffset,
32 const Datatype attrType) {
33 // initialize variables
34 bufMgr = bufMgrIn;
35 // Creating the index file name, taken from the project specification
36 std ::ostringstream indexStr;
37 indexStr << relationName << '.' << attrByteOffset;
38 outIndexName = indexStr.str();
39 // Since meta contains information about the first page, we have to get the header page
40 // but also have to see if the file exists.
41 try {
42 Page *headerPage;
43 file = new BlobFile(outIndexName, false);
44 headerPageNum = file->getFirstPageNo();
45
46 // After we get the first page, we use meta's info to compare with the given info to see if it matches.
47 bufMgr->readPage(file, headerPageNum, headerPage);
48 IndexMetaInfo *meta = (IndexMetaInfo *)headerPage;
49 if (relationName != meta->relationName) throw BadIndexInfoException("Index doesn't exist.");
50 if (attributeType != meta->attrType) throw BadIndexInfoException("Index doesn't exist.");
51 if (attrByteOffset != meta->attrByteOffset) throw BadIndexInfoException("Index doesn't exist.");
52 } catch (FileNotFoundException) { // This means file doesn't exist so create a file.
53
54 file = new BlobFile(outIndexName, true);
55 FileScan FS(relationName, bufMgr);
56 // get every tuple from the relation and load into the new file.
57 while (1) { // get all rids that meet the predicate and break if the end of the file of relationName is reached.
58 try {
59 RecordId rid;
60 // get
61 FS.scanNext(rid);
62 insertEntry(FS.getRecord().c_str(), rid);
63 } catch (EndOfFileException EOFE) {
64 break;
65 }
66 }
67 }
68 // bufMgrIn->readPage()
69 // if (relationName != meta)
70}
