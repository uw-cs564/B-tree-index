34void FileScan::scanNext(RecordId &outRid) {
35 std::string rec;
36
37 if (filePageIter == file->end()) {
38 throw EndOfFileException();
39 }
40
41 // special case of the first record of the first page of the file
42 if (curPage == NULL) {
43 // need to get the first page of the file
44 filePageIter = file->begin();
45 if (filePageIter == file->end()) {
46 throw EndOfFileException();
47 }
48
49 // read the first page of the file
50 bufMgr->readPage(file, (*filePageIter).page_number(), curPage);
51 curDirtyFlag = false;
52
53 // get the first record off the page
54 pageRecordIter = curPage->begin();
55
56 if (pageRecordIter != curPage->end()) {
57 // get pointer to record
58 rec = *pageRecordIter;
59
60 outRid = pageRecordIter.getCurrentRecord();
61 return;
62 }
63 }
64
65 // Loop, looking for a record that satisfied the predicate.
66 // First try and get the next record off the current page
67 pageRecordIter++;
68
69 while (pageRecordIter == curPage->end()) {
70 // unpin the current page
71 bufMgr->unPinPage(file, (*filePageIter).page_number(), curDirtyFlag);
72 curPage = NULL;
73 curDirtyFlag = false;
74
75 filePageIter++;
76 if (filePageIter == file->end()) {
77 curPage = NULL;
78 throw EndOfFileException();
79 }
80
81 // read the next page of the file
82 bufMgr->readPage(file, (*filePageIter).page_number(), curPage);
83
84 // get the first record off the page
85 pageRecordIter = curPage->begin();
86 }
87
88 // curRec points at a valid record
89 // see if the record satisfies the scan's predicate
90 // get a pointer to the record
91 rec = *pageRecordIter;
92
93 // return rid of the record
94 outRid = pageRecordIter.getCurrentRecord();
95 return;
96}
