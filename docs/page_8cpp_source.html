55void Page::updateRecord(const RecordId& record_id,
56 const std::string& record_data) {
57 validateRecordId(record_id);
58 const PageSlot* slot = getSlot(record_id.slot_number);
59 const std::size_t free_space_after_delete =
60 getFreeSpace() + slot->item_length;
61 if (record_data.length() > free_space_after_delete) {
62 throw InsufficientSpaceException(
63 page_number(), record_data.length(), free_space_after_delete);
64 }
65 // We have to disallow slot compaction here because we're going to place the
66 // record data in the same slot, and compaction might delete the slot if we
67 // permit it.
68 deleteRecord(record_id, false /* allow_slot_compaction */);
69 insertRecordInSlot(record_id.slot_number, record_data);
70}

76void Page::deleteRecord(const RecordId& record_id,
77 const bool allow_slot_compaction) {
78 validateRecordId(record_id);
79 PageSlot* slot = getSlot(record_id.slot_number);
80
81 for (int i = 0; i < slot->item_length; i++)
82 data_[i + slot->item_offset] = '\0';
83
84 //data_.replace(slot->item_offset, slot->item_length, slot->item_length, '\0');
85
86 // Compact the data by removing the hole left by this record (if necessary).
87 std::uint16_t move_offset = slot->item_offset;
88 std::size_t move_bytes = 0;
89 for (SlotId i = 1; i <= header_.num_slots; ++i) {
90 PageSlot* other_slot = getSlot(i);
91 if (other_slot->used && other_slot->item_offset < slot->item_offset) {
92 if (other_slot->item_offset < move_offset) {
93 move_offset = other_slot->item_offset;
94 }
95 move_bytes += other_slot->item_length;
96 // Update the slot for the other data to reflect the soon-to-be-new
97 // location.
98 other_slot->item_offset += slot->item_length;
99 }
100 }
101 // If we have data to move, shift it to the right.
102 if (move_bytes > 0) {
103 const std::string& data_to_move = std::string(data_, DATA_SIZE).substr(move_offset, move_bytes);
104
105 for (std::uint16_t i = 0; i < move_bytes; i++)
106 data_[i + move_offset + slot->item_length] = data_to_move[i];
107
108 //data_.replace(move_offset + slot->item_length, move_bytes, data_to_move);
109 }
110 header_.free_space_upper_bound += slot->item_length;
111
112 // Mark slot as unused.
113 slot->used = false;
114 slot->item_offset = 0;
115 slot->item_length = 0;
116 ++header_.num_free_slots;
117
118 if (allow_slot_compaction && record_id.slot_number == header_.num_slots) {
119 // Last slot in the list, so we need to free any unused slots that are at
120 // the end of the slot list.
121 int num_slots_to_delete = 1;
122 for (SlotId i = 1; i < header_.num_slots; ++i) {
123 // Traverse list backwards, looking for unused slots.
124 const PageSlot* other_slot = getSlot(header_.num_slots - i);
125 if (!other_slot->used) {
126 ++num_slots_to_delete;
127 } else {
128 // Stop at the first used slot we find, since we can't move used slots
129 // without affecting record IDs.
130 break;
131 }
132 }
133 header_.num_slots -= num_slots_to_delete;
134 header_.num_free_slots -= num_slots_to_delete;
135 header_.free_space_lower_bound -= sizeof(PageSlot) * num_slots_to_delete;
136 }
137}

155SlotId Page::getAvailableSlot() {
156 SlotId slot_number = INVALID_SLOT;
157 if (header_.num_free_slots > 0) {
158 // Have an allocated but unused slot that we can reuse.
159 for (SlotId i = 1; i <= header_.num_slots; ++i) {
160 const PageSlot* slot = getSlot(i);
161 if (!slot->used) {
162 // We don't decrement the number of free slots until someone actually
163 // puts data in the slot.
164 slot_number = i;
165 break;
166 }
167 }
168 } else {
169 // Have to allocate a new slot.
170 slot_number = header_.num_slots + 1;
171 ++header_.num_slots;
172 ++header_.num_free_slots;
173 header_.free_space_lower_bound = sizeof(PageSlot) * header_.num_slots;
174 }
175 assert(slot_number != INVALID_SLOT);
176 return slot_number;
177}

179void Page::insertRecordInSlot(const SlotId slot_number,
180 const std::string& record_data) {
181 if (slot_number > header_.num_slots ||
182 slot_number == INVALID_SLOT) {
183 throw InvalidSlotException(page_number(), slot_number);
184 }
185 PageSlot* slot = getSlot(slot_number);
186 if (slot->used) {
187 throw SlotInUseException(page_number(), slot_number);
188 }
189 const int record_length = record_data.length();
190 slot->used = true;
191 slot->item_length = record_length;
192 slot->item_offset = header_.free_space_upper_bound - record_length;
193 header_.free_space_upper_bound = slot->item_offset;
194 --header_.num_free_slots;
195
196 for (int i = 0; i < slot->item_length; i++)
197 data_[i + slot->item_offset] = record_data[i];
198
199 //data_.replace(slot->item_offset, slot->item_length, record_data);
200}
