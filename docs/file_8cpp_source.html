76void File::openIfNeeded(const bool create_new) {
77 if (open_counts_.find(filename_) != open_counts_.end()) { // exists an entry already
78 ++open_counts_[filename_];
79 stream_ = open_streams_[filename_];
80 } else {
81 std::ios_base::openmode mode =
82 std::fstream::in | std::fstream::out | std::fstream::binary;
83 const bool already_exists = exists(filename_);
84 if (create_new) {
85 // Error if we try to overwrite an existing file.
86 if (already_exists) {
87 throw FileExistsException(filename_);
88 }
89 // New files have to be truncated on open.
90 mode = mode | std::fstream::trunc;
91 } else {
92 // Error if we try to open a file that doesn't exist.
93 if (!already_exists) {
94 throw FileNotFoundException(filename_);
95 }
96 }
97 stream_.reset(new std::fstream(filename_, mode));
98 open_streams_[filename_] = stream_;
99 open_counts_[filename_] = 1;
100 }
101}

103void File::close() {
104 if (open_counts_[filename_] > 0)
105 --open_counts_[filename_];
106
107 stream_.reset();
108 assert(open_counts_[filename_] >= 0);

110 if (open_counts_[filename_] == 0) {
111 open_streams_.erase(filename_);
112 open_counts_.erase(filename_);
113 }

157Page PageFile::allocatePage(PageId& new_page_number) {
158 FileHeader header = readHeader();
159 Page new_page;
160 Page existing_page;
161 if (header.num_free_pages > 0) {
162 new_page = readPage(header.first_free_page, true /* allow_free */);
163 new_page.set_page_number(header.first_free_page);
164 new_page_number = new_page.page_number();
165 header.first_free_page = new_page.next_page_number();
166 --header.num_free_pages;
167
168 if (header.first_used_page == Page::INVALID_NUMBER ||
169 header.first_used_page > new_page.page_number()) {
170 // Either have no pages used or the head of the used list is a page later
171 // than the one we just allocated, so add the new page to the head.
172 if (header.first_used_page > new_page.page_number()) {
173 new_page.set_next_page_number(header.first_used_page);
174 }
175 header.first_used_page = new_page.page_number();
176 } else {
177 // New page is reused from somewhere after the beginning, so we need to
178 // find where in the used list to insert it.
179 PageId next_page_number = Page::INVALID_NUMBER;
180 for (FileIterator iter = begin(); iter != end(); ++iter) {
181 next_page_number = (*iter).next_page_number();
182 if (next_page_number > new_page.page_number() ||
183 next_page_number == Page::INVALID_NUMBER) {
184 existing_page = *iter;
185 break;
186 }
187 }
188 existing_page.set_next_page_number(new_page.page_number());
189 new_page.set_next_page_number(next_page_number);
190 }
191
192 assert((header.num_free_pages == 0) ==
193 (header.first_free_page == Page::INVALID_NUMBER));
194 } else {
195 new_page.set_page_number(header.num_pages);
196 new_page_number = new_page.page_number();
197
198 if (header.first_used_page == Page::INVALID_NUMBER) {
199 header.first_used_page = new_page.page_number();
200 } else {
201 // If we have pages allocated, we need to add the new page to the tail
202 // of the linked list.
203 for (FileIterator iter = begin(); iter != end(); ++iter) {
204 if ((*iter).next_page_number() == Page::INVALID_NUMBER) {
205 existing_page = *iter;
206 break;
207 }
208 }
209 assert(existing_page.isUsed());
210 existing_page.set_next_page_number(new_page.page_number());
211 }
212 ++header.num_pages;
213 }
214 writePage(new_page_number, new_page.header_, new_page);
215 if (existing_page.page_number() != Page::INVALID_NUMBER) {
216 // If we updated an existing page by inserting the new page into the
217 // used list, we need to write it out.
218 writePage(existing_page.page_number(), existing_page.header_, existing_page);
219 }
220 writeHeader(header);
221
222 return new_page;
223}

246void PageFile::writePage(const PageId new_page_number, const Page& new_page) {
247 PageHeader header = readPageHeader(new_page_number);
248 if (header.current_page_number == Page::INVALID_NUMBER) {
249 // Page has been deleted since it was read.
250 throw InvalidPageException(new_page_number, filename_);
251 }
252 // Page on disk may have had its next page pointer updated since it was read;
253 // we don't modify that, but we do keep all the other modifications to the
254 // page header.
255 const PageId next_page_number = header.next_page_number;
256 header = new_page.header_;
257 header.next_page_number = next_page_number;
258 writePage(new_page_number, header, new_page);
259}

261void PageFile::deletePage(const PageId page_number) {
262 FileHeader header = readHeader();
263
264 Page existing_page = readPage(page_number);
265 Page previous_page;
266 // If this page is the head of the used list, update the header to point to
267 // the next page in line.
268 if (page_number == header.first_used_page) {
269 header.first_used_page = existing_page.next_page_number();
270 } else {
271 // Walk the used list so we can update the page that points to this one.
272 for (FileIterator iter = begin(); iter != end(); ++iter) {
273 previous_page = *iter;
274 if (previous_page.next_page_number() == existing_page.page_number()) {
275 previous_page.set_next_page_number(existing_page.next_page_number());
276 break;
277 }
278 }
279 }
280 // Clear the page and add it to the head of the free list.
281 existing_page.initialize();
282 existing_page.set_next_page_number(header.first_free_page);
283 header.first_free_page = page_number;
284 ++header.num_free_pages;
285 if (previous_page.isUsed()) {
286 writePage(previous_page.page_number(), previous_page.header_, previous_page);
287 }
288 writePage(page_number, existing_page.header_, existing_page);
289 writeHeader(header);
290}
